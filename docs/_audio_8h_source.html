574 const XMVECTOR forward = XMVector3Rotate(g_XMIdentityR2, quat);
575 XMStoreFloat3(reinterpret_cast<XMFLOAT3*>(&OrientFront), forward);

577 const XMVECTOR up = XMVector3Rotate(g_XMIdentityR1, quat);
578 XMStoreFloat3(reinterpret_cast<XMFLOAT3*>(&OrientTop), up);
579 }
580
581 // Updates velocity and orientation by tracking changes in position over time.
582 void XM_CALLCONV Update(FXMVECTOR newPos, XMVECTOR upDir, float dt) noexcept
583 {
584 if (dt > 0.f)
585 {
586 const XMVECTOR lastPos = XMLoadFloat3(reinterpret_cast<const XMFLOAT3*>(&Position));
587
588 XMVECTOR vDelta = XMVectorSubtract(newPos, lastPos);
589 const XMVECTOR vt = XMVectorReplicate(dt);
590 XMVECTOR v = XMVectorDivide(vDelta, vt);
591 XMStoreFloat3(reinterpret_cast<XMFLOAT3*>(&Velocity), v);
592
593 vDelta = XMVector3Normalize(vDelta);
594 XMStoreFloat3(reinterpret_cast<XMFLOAT3*>(&OrientFront), vDelta);
595
596 v = XMVector3Cross(upDir, vDelta);
597 v = XMVector3Normalize(v);
598
599 v = XMVector3Cross(vDelta, v);
600 v = XMVector3Normalize(v);
601 XMStoreFloat3(reinterpret_cast<XMFLOAT3*>(&OrientTop), v);
602
603 XMStoreFloat3(reinterpret_cast<XMFLOAT3*>(&Position), newPos);
604 }
605 }

609 pCone = nullptr;
610 }
