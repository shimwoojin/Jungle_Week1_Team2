114 DIRECTX_TOOLKIT_API inline D3D12_GPU_DESCRIPTOR_HANDLE GetGpuHandle(_In_ size_t index) const
115 {
116 assert(m_pHeap != nullptr);
117 if (index >= m_desc.NumDescriptors)
118 {
119 throw std::out_of_range("D3DX12_GPU_DESCRIPTOR_HANDLE");
120 }
121 assert(m_desc.Flags & D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE);
122
123 D3D12_GPU_DESCRIPTOR_HANDLE handle;
124 handle.ptr = m_hGPU.ptr + UINT64(index) * UINT64(m_increment);
125 return handle;
126 }
