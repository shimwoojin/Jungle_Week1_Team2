197 // Shorthand for creating a root signature
198 DIRECTX_TOOLKIT_API
199 inline HRESULT CreateRootSignature(
200 _In_ ID3D12Device* device,
201 _In_ const D3D12_ROOT_SIGNATURE_DESC* rootSignatureDesc,
202 _Out_ ID3D12RootSignature** rootSignature) noexcept
203 {
204 if (!device || !rootSignatureDesc || !rootSignature)
205 return E_INVALIDARG;
206
207 Microsoft::WRL::ComPtr<ID3DBlob> pSignature;
208 Microsoft::WRL::ComPtr<ID3DBlob> pError;
209 HRESULT hr = D3D12SerializeRootSignature(rootSignatureDesc, D3D_ROOT_SIGNATURE_VERSION_1, pSignature.GetAddressOf(), pError.GetAddressOf());
210 if (SUCCEEDED(hr))
211 {
212 hr = device->CreateRootSignature(0, pSignature->GetBufferPointer(), pSignature->GetBufferSize(),
213 IID_GRAPHICS_PPV_ARGS(rootSignature)
214 );
215 }
216 return hr;
217 }

219 // Helper for obtaining texture size
220 DIRECTX_TOOLKIT_API
221 inline XMUINT2 GetTextureSize(_In_ ID3D12Resource* tex) noexcept
222 {
223 if (!tex)
224 return XMUINT2(0, 0);
225
226 #if defined(_MSC_VER) || !defined(_WIN32)
227 const auto desc = tex->GetDesc();
228 #else
229 D3D12_RESOURCE_DESC tmpDesc;
230 const auto& desc = *tex->GetDesc(&tmpDesc);
231 #endif
232 return XMUINT2(static_cast<uint32_t>(desc.Width), static_cast<uint32_t>(desc.Height));
233 }

283// Helper for resource barrier.
284 DIRECTX_TOOLKIT_API
285 inline void TransitionResource(
286 _In_ ID3D12GraphicsCommandList* commandList,
287 _In_ ID3D12Resource* resource,
288 D3D12_RESOURCE_STATES stateBefore,
289 D3D12_RESOURCE_STATES stateAfter) noexcept
290 {
291 assert(commandList != nullptr);
292 assert(resource != nullptr);
293
294 if (stateBefore == stateAfter)
295 return;
296
297 D3D12_RESOURCE_BARRIER desc = {};
298 desc.Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
299 desc.Transition.pResource = resource;
300 desc.Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
301 desc.Transition.StateBefore = stateBefore;
302 desc.Transition.StateAfter = stateAfter;
303
304 commandList->ResourceBarrier(1, &desc);
305 }

307 // Helper which applies one or more resources barriers and then reverses them on destruction.
308 class DIRECTX_TOOLKIT_API ScopedBarrier
309 {
310 public:
311 ScopedBarrier(
312 _In_ ID3D12GraphicsCommandList* commandList,
313 std::initializer_list<D3D12_RESOURCE_BARRIER> barriers) noexcept(false)
314 : mCommandList(commandList),
315 mBarriers(barriers)
316 {
317 assert(mBarriers.size() <= UINT32_MAX);
318
319 // Set barriers
320 mCommandList->ResourceBarrier(static_cast<UINT>(mBarriers.size()), mBarriers.data());
321 }
322
323 ScopedBarrier(
324 _In_ ID3D12GraphicsCommandList* commandList,
325 _In_reads_(count) const D3D12_RESOURCE_BARRIER *barriers,
326 size_t count) noexcept(false)
327 : mCommandList(commandList),
328 mBarriers(barriers, barriers + count)
329 {
330 assert(count <= UINT32_MAX);
331
332 // Set barriers
333 mCommandList->ResourceBarrier(static_cast<UINT>(mBarriers.size()), mBarriers.data());
334 }
335
336 template<size_t TBarrierLength>
337 ScopedBarrier(
338 _In_ ID3D12GraphicsCommandList* commandList,
339 const D3D12_RESOURCE_BARRIER(&barriers)[TBarrierLength]) noexcept(false)
340 : mCommandList(commandList),
341 mBarriers(barriers, barriers + TBarrierLength)
342 {
343 assert(TBarrierLength <= UINT32_MAX);
344
345 // Set barriers
346 mCommandList->ResourceBarrier(static_cast<UINT>(mBarriers.size()), mBarriers.data());
347 }
348
349 ScopedBarrier(ScopedBarrier&&) = default;
350 ScopedBarrier& operator= (ScopedBarrier&&) = default;
351
352 ScopedBarrier(ScopedBarrier const&) = delete;
353 ScopedBarrier& operator= (ScopedBarrier const&) = delete;
354
355 ~ScopedBarrier()
356 {
357 // reverse barrier inputs and outputs
358 for (auto& b : mBarriers)
359 {
360 std::swap(b.Transition.StateAfter, b.Transition.StateBefore);
361 }
362
363 // Set barriers
364 mCommandList->ResourceBarrier(static_cast<UINT>(mBarriers.size()), mBarriers.data());
365 }
366
367 private:
368 ID3D12GraphicsCommandList* mCommandList;
369 std::vector<D3D12_RESOURCE_BARRIER> mBarriers;
370 };

374 // Helper to check for power-of-2
375 template<typename T>
376 constexpr bool IsPowerOf2(T x) noexcept { return ((x != 0) && !(x & (x - 1))); }
377
378 // Helpers for aligning values by a power of 2
379 template<typename T>
380 inline T AlignDown(T size, size_t alignment) noexcept
381 {
382 if (alignment > 0)
383 {
384 assert(((alignment - 1) & alignment) == 0);
385 auto mask = static_cast<T>(alignment - 1);
386 return size & ~mask;
387 }
388 return size;
389 }
390
391 template<typename T>
392 inline T AlignUp(T size, size_t alignment) noexcept
393 {
394 if (alignment > 0)
395 {
396 assert(((alignment - 1) & alignment) == 0);
397 auto mask = static_cast<T>(alignment - 1);
398 return (size + mask) & ~mask;
399 }
400 return size;
401 }
402 }
