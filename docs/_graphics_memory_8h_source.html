186 // Make sure to keep the GraphicsResource handle alive as long as you need to access
187 // the memory on the CPU. For example, do not simply cache GpuAddress() and discard
188 // the GraphicsResource object, or your memory may be overwritten later.
189 DIRECTX_TOOLKIT_API inline GraphicsResource __cdecl Allocate(size_t size, size_t alignment = 16, uint32_t tag = TAG_GENERIC)
190 {
191 auto alloc = AllocateImpl(size, alignment);
192 #ifdef USING_PIX_CUSTOM_MEMORY_EVENTS
193 std::ignore = ReportCustomMemoryAlloc(alloc.Memory(), alloc.Size(), tag);
194 #else
195 UNREFERENCED_PARAMETER(tag);
196 #endif
197 return alloc;
198 }
199
200 // Version of Allocate that aligns to D3D12 constant buffer requirements
201 template<typename T> GraphicsResource AllocateConstant()
202 {
203 constexpr size_t alignment = D3D12_CONSTANT_BUFFER_DATA_PLACEMENT_ALIGNMENT;
204 constexpr size_t alignedSize = (sizeof(T) + alignment - 1) & ~(alignment - 1);
205 auto alloc = AllocateImpl(alignedSize, alignment);
206 #ifdef USING_PIX_CUSTOM_MEMORY_EVENTS
207 // This cast is needed to capture the type information in the PDB
208 std::ignore = reinterpret_cast<T*>(ReportCustomMemoryAlloc(alloc.Memory(), alloc.Size(), TAG_CONSTANT));
209 #endif
210 return alloc;
211 }
212 template<typename T> GraphicsResource AllocateConstant(const T& setData)
213 {
214 GraphicsResource alloc = AllocateConstant<T>();
215 memcpy(alloc.Memory(), &setData, sizeof(T));
216 return alloc;
217 }
