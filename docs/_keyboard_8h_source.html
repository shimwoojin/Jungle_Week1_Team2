456 bool __cdecl IsKeyDown(Keys key) const noexcept
457 {
458 if (key <= 0xfe)
459 {
460 auto ptr = reinterpret_cast<const uint32_t*>(this);
461 const unsigned int bf = 1u << (key & 0x1f);
462 return (ptr[(key >> 5)] & bf) != 0;
463 }
464 return false;
465 }
466
467 bool __cdecl IsKeyUp(Keys key) const noexcept
468 {
469 if (key <= 0xfe)
470 {
471 auto ptr = reinterpret_cast<const uint32_t*>(this);
472 const unsigned int bf = 1u << (key & 0x1f);
473 return (ptr[(key >> 5)] & bf) == 0;
474 }
475 return false;
476 }
