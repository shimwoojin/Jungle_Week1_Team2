183 // Draw the mesh with a range of effects that mesh parts will index into.
184 // Effects can be any IEffect pointer type (including smart pointer). Value or reference types will not compile.
185 // The iterator passed to this method should have random access capabilities for best performance.
186 template<typename TEffectIterator, typename TEffectIteratorCategory = typename TEffectIterator::iterator_category>
187 static void DrawMeshParts(
188 _In_ ID3D12GraphicsCommandList* commandList,
189 const Collection& meshParts,
190 TEffectIterator partEffects)
191 {
192 // This assert is here to prevent accidental use of containers that would cause undesirable performance penalties.
193 static_assert(
194 std::is_base_of<std::random_access_iterator_tag, TEffectIteratorCategory>::value,
195 "Providing an iterator without random access capabilities -- such as from std::list -- is not supported.");
196
197 for (const auto& it : meshParts)
198 {
199 auto part = it.get();
200 assert(part != nullptr);
201
202 // Get the effect at the location specified by the part's material
203 TEffectIterator effect_iterator = partEffects;
204 std::advance(effect_iterator, part->partIndex);
205
206 // Apply the effect and draw
207 (*effect_iterator)->Apply(commandList);
208 part->Draw(commandList);
209 }
210 }
211
212 template<typename TEffectIterator, typename TEffectIteratorCategory = typename TEffectIterator::iterator_category>
213 static void XM_CALLCONV DrawMeshParts(
214 _In_ ID3D12GraphicsCommandList* commandList,
215 const Collection& meshParts,
216 FXMMATRIX world,
217 TEffectIterator partEffects)
218 {
219 // This assert is here to prevent accidental use of containers that would cause undesirable performance penalties.
220 static_assert(
221 std::is_base_of<std::random_access_iterator_tag, TEffectIteratorCategory>::value,
222 "Providing an iterator without random access capabilities -- such as from std::list -- is not supported.");
223
224 for (const auto& it : meshParts)
225 {
226 auto part = it.get();
227 assert(part != nullptr);
228
229 // Get the effect at the location specified by the part's material
230 TEffectIterator effect_iterator = partEffects;
231 std::advance(effect_iterator, part->partIndex);
232
233 auto imatrices = dynamic_cast<IEffectMatrices*>((*effect_iterator).get());
234 if (imatrices)
235 {
236 imatrices->SetWorld(world);
237 }
238
239 // Apply the effect and draw
240 (*effect_iterator)->Apply(commandList);
241 part->Draw(commandList);
242 }
243 }

578 // Utility function for getting a GPU descriptor for a mesh part/material index. If there is no texture the
579 // descriptor will be zero.
580 D3D12_GPU_DESCRIPTOR_HANDLE __cdecl GetGpuTextureHandleForMaterialIndex(uint32_t materialIndex, _In_ ID3D12DescriptorHeap* heap, _In_ size_t descriptorSize, _In_ size_t descriptorOffset) const
581 {
582 D3D12_GPU_DESCRIPTOR_HANDLE handle = {};
583
584 if (materialIndex >= materials.size())
585 return handle;
586
587 const int textureIndex = materials[materialIndex].diffuseTextureIndex;
588 if (textureIndex == -1)
589 return handle;
590
591 #if defined(_MSC_VER) || !defined(_WIN32)
592 handle = heap->GetGPUDescriptorHandleForHeapStart();
593 #else
594 std::ignore = heap->GetGPUDescriptorHandleForHeapStart(&handle);
595 #endif
596 handle.ptr += static_cast<UINT64>(descriptorSize * (UINT64(textureIndex) + UINT64(descriptorOffset)));
597
598 return handle;
599 }

669 void XM_CALLCONV ModelMeshPart::DrawSkinnedMeshParts(
670 _In_ ID3D12GraphicsCommandList* commandList,
671 const ModelMesh& mesh,
672 const ModelMeshPart::Collection& meshParts,
673 size_t nbones,
674 _In_reads_(nbones) const XMMATRIX* boneTransforms,
675 FXMMATRIX world,
676 TEffectIterator partEffects)
677 {
678 // This assert is here to prevent accidental use of containers that would cause undesirable performance penalties.
679 static_assert(
680 std::is_base_of<std::random_access_iterator_tag, TEffectIteratorCategory>::value,
681 "Providing an iterator without random access capabilities -- such as from std::list -- is not supported.");
682
683 assert(nbones > 0 && boneTransforms != nullptr);
684
685 ModelBone::TransformArray temp;
686
687 for (const auto& mit : meshParts)
688 {
689 auto part = mit.get();
690 assert(part != nullptr);
691
692 // Get the effect at the location specified by the part's material
693 TEffectIterator effect_iterator = partEffects;
694 std::advance(effect_iterator, part->partIndex);
695
696 auto imatrices = dynamic_cast<IEffectMatrices*>((*effect_iterator).get());
697 if (imatrices)
698 {
699 imatrices->SetWorld(world);
700 }
701
702 auto iskinning = dynamic_cast<IEffectSkinning*>((*effect_iterator).get());
703 if (iskinning)
704 {
705 if (mesh.boneInfluences.empty())
706 {
707 // Direct-mapping of vertex bone indices to our master bone array
708 iskinning->SetBoneTransforms(boneTransforms, nbones);
709 }
710 else
711 {
712 if (!temp)
713 {
714 // Create the influence mapped bones on-demand.
715 temp = ModelBone::MakeArray(IEffectSkinning::MaxBones);
716
717 size_t count = 0;
718 for (auto it : mesh.boneInfluences)
719 {
720 ++count;
721 if (count > IEffectSkinning::MaxBones)
722 {
723 throw std::runtime_error("Too many bones for skinning");
724 }
725
726 if (it >= nbones)
727 {
728 throw std::runtime_error("Invalid bone influence index");
729 }
730
731 temp[count - 1] = boneTransforms[it];
732 }
733
734 assert(count == mesh.boneInfluences.size());
735 }
736
737 iskinning->SetBoneTransforms(temp.get(), mesh.boneInfluences.size());
738 }
739 }
740 else if (imatrices)
741 {
742 // Fallback for if we encounter a non-skinning effect in the model
743 XMMATRIX bm = (mesh.boneIndex != ModelBone::c_Invalid && mesh.boneIndex < nbones)
744 ? boneTransforms[mesh.boneIndex] : XMMatrixIdentity();
745
746 imatrices->SetWorld(XMMatrixMultiply(bm, world));
747 }
748
749 // Apply the effect and draw
750 (*effect_iterator)->Apply(commandList);
751 part->Draw(commandList);
752 }
753 }
