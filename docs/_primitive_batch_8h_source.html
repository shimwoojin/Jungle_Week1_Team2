109 // Similar to the D3D9 API DrawPrimitiveUP.
110 void Draw(
111 D3D_PRIMITIVE_TOPOLOGY topology,
112 _In_reads_(vertexCount) TVertex const* vertices, size_t vertexCount)
113 {
114 void* mappedVertices;
115
116 PrimitiveBatchBase::Draw(topology, false, nullptr, 0, vertexCount, &mappedVertices);
117
118 memcpy(mappedVertices, vertices, vertexCount * sizeof(TVertex));
119 }
